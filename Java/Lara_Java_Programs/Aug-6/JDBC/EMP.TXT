DROP SEQUENCE EMPLOYEE_SEQ;
CREATE SEQUENCE EMPLOYEE_SEQ INCREMENT BY 1;

BROP TABLE EMPLOYEE;

CREATE TABLE EMPLOYEE(
			ID NUMBER,FIRST_NAME VARCHAR2(10),
			LAST_NAME VARCHAR2(10),
			DATE_BIRTH TIMESTAMP,
			JOINING_DATE DATE,
			USERNAME VARCHAR2(10),
			PASSWORD VARCHAR2(10),
			GENDER_ID NUMBER,
			LATEST_EDUCATION_ID NUMBER,
			PROJECT_ID NUMBER,
CONSTRAINT EMPLOYEE_PK PRIMARY KEY(ID),
CONSTRAINT EMPLOYEE_FK1 FOREIGN KEY(GENDER_ID) REFERENCES GENDER(ID),
CONSTRAINT EMPLOYEE_FK2 FOREIGN KEY(LATEST_EDUCATION_ID) REFERENCES EDUCATION(ID),
CONSTRAINT EMPLOYEE_FK2 FOREIGN KEY(PROJECT_ID) REFERENCES PROJECT(ID));
